Codeforces 1334E-Divisor Paths

发布于 2020-04-12 | 5分钟 | 904字数

我们发现，每次走过一条路径都是减少/增加一个质因子。而且边权是减少/增加一个质因子后减少/增加的因数个数。那么显然有：x→yx\to yx→y的最优路径一定是x→gcd(x,y)→yx\to gcd(x,y)\to yx→gcd(x,y)→y。因为这样能保证减少/增加的质因子是其他所有方案减少/增加的质因子的子集，那么显然减少/增加的因数个数是最少的。

又因为x→gcd(x,y)x\to gcd(x,y)x→gcd(x,y)的路径上，因数一定是不停减少，不会增加的。而无论删除质因子的顺序如何，减少的因数集合一定是固定不变的。所以删去质因子的顺序可以任意排列。gcd(x,y)→ygcd(x,y)\to ygcd(x,y)→y的路径同理。

那么算法就很显然了。我们先分解DDD，然后将xgcd(x,y)\frac{x}{gcd(x,y)}gcd(x,y)x​质因数分解。（因为xgcd(x,y)\frac{x}{gcd(x,y)}gcd(x,y)x​包含的质因子一定是DDD的质因子，所以我们可以直接利用先前分解好的DDD的质因子一个一个尝试即可）我们只需要知道xgcd(x,y)\frac{x}{gcd(x,y)}gcd(x,y)x​质因数分解出的质因子个数cntcntcnt即可，这样一次分解利用之前的预处理数据后复杂度是log⁡D\log DlogD。证明显然。

那么x→gcd(x,y)x\to gcd(x,y)x→gcd(x,y)的方案数就是cnt!∏v is prime cntv!\frac{cnt!}{\prod _{\text{v is prime }}cnt_v!}∏v is prime ​cntv​!cnt!​。gcd(x,y)→ygcd(x,y)\to ygcd(x,y)→y的方案数同理。然后乘在一起就好了。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<vector>
#include<queue>

#define mk make_pair

#define rep(i,a,b) for(int i=(a),i##end=(b);i<=i##end;i++)
#define fi first
#define se second
typedef long long ll;
using namespace std;
const int maxn=3e5+100;
const int mod=998244353;
ll d,q;
ll prime[maxn],prime_cnt,frac[maxn];
ll gcd(ll a,ll b){return b?gcd(b,a%b):a;}
int ksm(ll num,ll t) {
    ll res=1;num%=mod;
    for(;t;t>>=1,num=num*num%mod) {
        if(t&1)res=num*res%mod;
    }
    return res%mod;
}
ll solve(ll u) {
    ll a=1,b=1,tot=0;//分子分母
    rep(i,1,prime_cnt) {
        ll cnt=0;while(u%prime[i]==0)u/=prime[i],cnt++;
        b=b*frac[cnt]%mod;tot+=cnt;
    }
    a=frac[tot];
    return a*ksm(b,mod-2)%mod;
}
int main() {
    ios::sync_with_stdio(0);
    cin>>d>>q;
    frac[0]=frac[1]=1;//预处理阶乘
    rep(i,2,100)frac[i]=frac[i-1]*i%mod;
    ll tmp=d;
    for(ll i=2;i*i<=d;i++) {
        if(tmp%i==0)prime[++prime_cnt]=i;
        while(tmp%i==0)tmp/=i;
    }
    if(tmp>1)prime[++prime_cnt]=tmp;
    rep(i,1,q) {
        ll u,v;cin>>u>>v;ll g=gcd(u,v);
        cout<<solve(u/g)*solve(v/g)%mod<<endl;
    }
    return 0;
}
